query Posts($limit: Float!, $cursor: String) {
  paginatedPosts(limit: $limit, cursor: $cursor) {
    posts {
      id
      body
      title
      createdAt
      creatorId
      owner
      likes {
        ...RegularUser
      }
      likeStatus
      creator {
        ...RegularUser
      }
    }
    hasMore
  }
}
